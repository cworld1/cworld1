export function Container({ color, children }) {
  return (
    <div
      className='f1-ct border rounded-xl p-4 flex flex-col gap-4'
      style={`background-color:${color}15; --team-color:${color}`}
    >
      {children}
      <style>{'.f1-ct p{margin:0};'}</style>
    </div>
  )
}

export function TeamTable({
  link,
  name,
  nickname,
  fullName,
  teamChief,
  reserve,
  year,
  podiums,
  color,
  children
}) {
  return (
    <div className='grid gap-4 md:grid-cols-2'>
      <div className='max-md:max-w-100'>{children}</div>
      <div className='flex flex-col gap-4'>
        <div className='max-md:flex flex-wrap items-end gap-x-2'>
          <div>
            <a className='text-foreground text-lg font-bold my-0' href={link} target='_blank'>
              {name}
            </a>
            {nickname && nickname !== '' && (
              <span className='text-sm text-muted-foreground ms-2 relative top-0.5 max-md:top-0.1'>
                {' '}
                [{nickname}]
              </span>
            )}
          </div>
          {fullName && <div className='text-muted-foreground my-0'>{fullName}</div>}
        </div>
        <div>
          <div className='flex flex-wrap gap-x-8 gap-y-2'>
            <div>
              <div className='text-xs text-muted-foreground'>领队</div>
              <div className='text-foreground mt-0'>{teamChief}</div>
            </div>
            <div>
              <div className='text-xs text-muted-foreground'>预备车手</div>
              <div className='text-foreground mt-0'>{reserve || '-'}</div>
            </div>
          </div>
          <div className='max-sm:flex flex-wrap gap-x-1 mt-2'>
            <span>{year} / </span>
            <span>{podiums} 次领奖 / </span>
            <span>{color}</span>
          </div>
        </div>
      </div>
    </div>
  )
}

export function DriversTable({ drivers, children }) {
  return (
    <div>
      {children}
      <div className='flex-1 pt-4 grid grid-cols-2 gap-x-4'>
        {drivers.map((driver, idx) => (
          <div key={idx} className='flex flex-col gap-y-2'>
            <div className='flex flex-wrap items-center gap-2'>
              <span
                className='font-mono rounded border px-1 max-md:text-sm md:px-1.5'
                style='border-color:var(--team-color);color:var(--team-color)'
              >
                {driver.abbr}
              </span>
              <div>
                <a className='text-foreground' href={driver.link} target='_blank'>
                  {driver.name}
                </a>
                {driver.nickname && driver.nickname != '' && (
                  <span className='text-sm text-muted-foreground ms-2 relative top-0.5'>
                    [{driver.nickname}]
                  </span>
                )}
              </div>
            </div>
            <div className='max-sm:flex flex-wrap gap-x-1'>
              <span>#{driver.num} / </span>
              <span>{new Date().getFullYear() - driver.birth} 岁 / </span>
              <span>{driver.nation} / </span>
              <span>{driver.podiums} 次领奖</span>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}

其实官网本来就有长期在做 [F1 Teams 2025](https://www.formula1.com/en/teams)，但是纯英文有时候会不那么对应的上（同样的还有缩写），并且网络大部份资料还较为老旧，这就由此引出了本文。简单声明：

1. 本文按照官网顺序依次介绍所有车队，确保没有隐性个人倾向。
2. 领奖（登上领奖台，Podiums）次数是截至 2025 年初的统计数据。
3. 车手年龄为按照当前年份减去出生年份实时计算。

那么祝大家查阅愉快。
